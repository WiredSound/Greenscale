121  // It takes a base stat (eg. the components max integrity) and then goes through all the upgrades the component has and applies them.
122  // The modifierFunc argument takes each component upgrade and returns the modifier that should be used (eg. upgrade.maxIntegrityModifier).
123  template <typename T>
124  T statWithUpgradesApplied(T baseValue, ModifierFunc modiferFunc) {
125  T value = baseValue;
126 
127  for (const ComponentUpgrade &upgrade : upgrades) {
128  auto increasePercent = modiferFunc(upgrade);
129  value += static_cast<T>(baseValue * increasePercent);
130  }
